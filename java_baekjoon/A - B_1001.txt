import java.util.Scanner;

public class Main {
	public static void main(String[] args) {
		Scanner scan = new Scanner(System.in);
		int a = scan.nextInt();
		int b = scan.nextInt();
		
		System.out.print(calculate(a, b));
		scan.close();
		
	}
	
	public static int calculate(int a, int b) {
		return a - b;
	}
}
